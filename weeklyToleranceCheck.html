```{r}
demfile$weight <- as.numeric(demfile$weight)
demfile$HHwgt <- as.numeric(demfile$HHweight)



demfile <- demfile %>%
  select(everything()) %>%
  group_by(date) %>%
  mutate(weight.norm = weight / mean(weight),
         week = lubridate::week(date)) %>% ungroup()

dayTab <-
  demfile %>%
  select(week, date, HHid, weight, weight.norm) %>%
  group_by(week, date) %>%
  summarise(
    HHs = round(n_distinct(HHid), digits = 0),
    "Sample size(N)" = round(n(), digits = 0),
    "Sum Weights" = round(sum(weight), digits = 2),
    "Mean Weight" = round(mean(weight), digits = 2),
    sd = round(sd(weight.norm), digits = 2),
    EFF = round((1/(1 + sd^2)), digits = 2)) %>% ungroup()
  


demfileHH <- demfile %>%
  select(week, date, HHid, HHwgt) %>%
  group_by(date) %>%
  filter(!duplicated(HHid)) %>% 
  mutate(HHwgt.norm = HHwgt/mean(HHwgt))

dayTabHH <- demfileHH %>% 
  group_by(week, date) %>% 
  summarise(
  HHsHH = round(n_distinct(HHid), digits = 0),
  sumHHwgt = round(sum(HHwgt), digits = 2),
  meanHHwgt = round(mean(HHwgt), digits = 2),
  HHsd = round(sd(HHwgt.norm), digits = 2),
  HH_EFF = round((1/(1 + HHsd^2)), digits = 2)) %>% ungroup()
  

dayTab <- left_join(dayTab, dayTabHH)

  


weekTab <- dayTab %>%
  group_by(week) %>%
  summarise(
    `Sample size(N)` = mean(`Sample size(N)`),
    HHs = round(mean(HHs), digits = 0),
    `Sum Weights` = mean(`Sum Weights`),
    `Sum HH Weights` = mean(`sumHHwgt`),
    sd = mean(sd),
    EFF = mean(EFF),
    HHsd = mean(HHsd),
    HH_EFF = mean(HH_EFF)
  )




kbl(
  weekTab,
  digits = c(0, 0, 0, 0, 2, 2, 2, 2, 2),
  booktabs = TRUE,
  caption = "KSA TAM. HH & Ind sample KPIs -  week 13/14 2023",
  format.args = list(decimal.mark = ".", big.mark = ",")
) %>% kable_styling(full_width = T, font_size = 10)
```
